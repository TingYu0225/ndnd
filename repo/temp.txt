// Create keychain and trust config
	store := storage.NewMemoryStore()
	kc, err := keychain.NewKeyChain(t.keychain_uri, store)
	if err != nil {
		fmt.Println("keychain error:", err)
		return
	}

	// Convert keychain_uri to file path and read root.cert
	// Handle dir:// URI format by stripping prefix
	kcPath := t.keychain_uri
	if strings.HasPrefix(kcPath, "dir://") {
		kcPath = kcPath[6:] // Remove "dir://"
	}

	rootCertPath := filepath.Join(kcPath, "root.cert")
	rootCertBytes, err := os.ReadFile(rootCertPath)
	if err != nil {
		fmt.Println("failed to read root cert from", rootCertPath, ":", err)
		return
	}

	// Insert root cert into keychain
	err = kc.InsertCert(rootCertBytes)
	if err != nil {
		fmt.Println("failed to insert root cert:", err)
		return
	}

	// Parse root cert to get its name
	rootCertData, _, err := spec.Spec{}.ReadData(enc.NewBufferView(rootCertBytes))
	if err != nil {
		fmt.Println("failed to parse root cert:", err)
		return
	}

	// Read and insert client cert
	clientCertPath := filepath.Join(kcPath, "client.cert")
	clientCertBytes, err := os.ReadFile(clientCertPath)
	if err != nil {
		fmt.Println("failed to read client cert from", clientCertPath, ":", err)
		return
	}

	err = kc.InsertCert(clientCertBytes)
	if err != nil {
		fmt.Println("failed to insert client cert:", err)
		return
	}
	fmt.Println("Client cert loaded for signing")

	// Create trust config with root cert as anchor
	// schema := trust_schema.NewLvsSchema(schemaBytes) // 改用實際驗證
	schema := trust_schema.NewNullSchema()
	trust, err := sec.NewTrustConfig(kc, schema, []enc.Name{rootCertData.Name()})
	if err != nil {
		fmt.Println("trust config creation failed:", err)
		return
	}
	fmt.Println("Trust config created with root anchor:", rootCertData.Name())
